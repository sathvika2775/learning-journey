class Solution {
    public int maxArea(int[] h) {
        int count =0;
        int i=0,j=h.length-1;
        while(i<j){
            int m = Math.min(h[i],h[j]);
            count=Math.max(count,m*(j-i));
            if(h[i]<h[j]){
                i++;
            }else{
                j--;
            }
        } 
        return count;
    }
}
