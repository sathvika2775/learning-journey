class Solution {
    public int majorityElement(int[] nums) {
       Map<Integer,Integer> m=new HashMap<>();
       for(int i=0;i<nums.length;i++){
            m.put(nums[i],m.getOrDefault(nums[i],0)+1);
       } 
       int max=0;
       int ans=0;
       for(int k:m.keySet()){
            if(max<m.get(k)){
                max=m.get(k);
                ans=k;
            }
       }
       return ans;
    }
}
