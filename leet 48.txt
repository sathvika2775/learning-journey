class Solution {
    public void rotate(int[][] matrix) {
        int n=matrix.length;
        int ans[][]=new int[n][n];
        int k=0;
        for(int j=0;j<n && k<n;j++){
            for(int i=n-1;i>=0;i--){
                ans[j][k++]=matrix[i][j];
            }
            k=0;
        }
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                matrix[i][j]=ans[i][j];
            }
        }
        return ;
    }
}
