class Solution {
    public int removeElement(int[] nums, int val) {
        List<Integer> l=new LinkedList<>();
        for(int ele:nums){
            if(ele!=val){
                l.add(ele);
            }
        }
        for(int i=0;i<l.size();i++){
            nums[i]=l.get(i);
        }
        return l.size();
    }
}
